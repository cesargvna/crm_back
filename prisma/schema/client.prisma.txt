model Client {
  id                String           @id @default(uuid()) @db.VarChar(255)
  name              String           @db.VarChar(20)
  status            Boolean          @default(true)
  client_points     Int              @default(0)

  lastname          String?          @db.VarChar(20)
  ci                String?          @db.VarChar(20)
  nit               String?          @db.VarChar(20)
  description       String?          @db.VarChar(100)
  address           String?          @db.VarChar(100)
  cellphone         String?          @db.VarChar(20)
  telephone         String?          @db.VarChar(20)
  email             String?          @db.VarChar(20)
  
  clientCategoryId  String           @db.VarChar(255)
  tenantId          String           @db.VarChar(255)
  subsidiaryId      String           @db.VarChar(255)

  created_at        DateTime         @default(now())
  updated_at        DateTime         @updatedAt

  clientCategory    ClientCategory   @relation(fields: [clientCategoryId], references: [id])
  sales             Sale[]

  @@unique([name, tenantId, subsidiaryId])
  @@index([tenantId])
  @@index([subsidiaryId])
}

model ClientCategory {
  id            String    @id @default(uuid()) @db.VarChar(255)
  name          String    @db.VarChar(50)
  status        Boolean   @default(true)
  description   String?   @db.VarChar(100)

  tenantId      String    @db.VarChar(255)
  subsidiaryId  String    @db.VarChar(255)

  created_at    DateTime  @default(now())
  updated_at    DateTime  @updatedAt

  clients       Client[]

  @@unique([name, tenantId, subsidiaryId])
  @@index([tenantId])
  @@index([subsidiaryId])
}
